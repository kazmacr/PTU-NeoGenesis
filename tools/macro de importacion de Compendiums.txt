new Dialog({
  title: "Importador Masivo de Especies (Neo Génesis)",
  content: `
    <p>1. Pega aquí el contenido de tu archivo JSON:</p>
    <textarea id="json-import-data" style="width: 100%; height: 250px; font-family: monospace;"></textarea>
    <br><br>
    <p>2. Escribe el ID del compendio destino (Ej: <b>world.pokedex</b>):</p>
    <input type="text" id="pack-id" value="world." />
    <br><br>
  `,
  buttons: {
    import: {
      label: "<i class='fas fa-file-import'></i> Importar Especies",
      callback: async (html) => {
        const jsonText = html.find("#json-import-data").val();
        const packId = html.find("#pack-id").val();
        const pack = game.packs.get(packId);
        
        if (!pack) {
            return ui.notifications.error(`No se encontró el compendio: ${packId}. Verifica el ID.`);
        }
        
        try {
          const data = JSON.parse(jsonText);
          // Desbloquear el compendio temporalmente si está bloqueado
          const wasLocked = pack.locked;
          if (wasLocked) await pack.configure({locked: false});
          
          // Crear los documentos
          await Item.createDocuments(data, { pack: pack.collection });
          
          // Volver a bloquear si estaba bloqueado
          if (wasLocked) await pack.configure({locked: true});
          
          ui.notifications.info(`¡Éxito! Se importaron ${data.length} Pokémon al compendio.`);
        } catch (e) {
          ui.notifications.error("Error al leer el JSON. Revisa la consola (F12) para más detalles.");
          console.error("Error de importación:", e);
        }
      }
    }
  },
  default: "import"
}).render(true);
